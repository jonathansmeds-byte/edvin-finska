function shuffle(arr){

  for(let i = arr.length - 1; i > 0; i--){

    const j = Math.floor(Math.random() * (i + 1));

    [arr[i], arr[j]] = [arr[j], arr[i]];

  }

  return arr;

}

 

// Normalisering för "snäll" rättning: trim, lowercase, collapse spaces, ta bort vissa skiljetecken.

function normalize(s){

  return (s ?? "")

    .toString()

    .trim()

    .toLowerCase()

    .replace(/[“”"'.!,?;:()[\]{}]/g, "")

    .replace(/\s+/g, " ");

}

 

// En enkel "nära"-koll: om normaliserat svar är substring eller har hög likhet via token-overlap.

function isNear(user, correct){

  const u = normalize(user);

  const c = normalize(correct);

 

  if(!u) return false;

  if(u === c) return true;

  if(c.includes(u) || u.includes(c)) return true;

 

  const ut = new Set(u.split(" "));

  const ct = new Set(c.split(" "));

  let inter = 0;

  for(const t of ut) if(ct.has(t)) inter++;

  const jaccard = inter / Math.max(1, new Set([...ut, ...ct]).size);

 

  return jaccard >= 0.6;

}

 

// Tillåt flera facitvarianter separerade med "/"

function splitAnswers(s){

  return (s ?? "").split("/").map(x => x.trim()).filter(Boolean);

}

 

function buildDeck(){

  const levelVal = levelSel.value;

  let items = [];

  if(levelVal === "mix12"){

    items = [...LEVELS[1], ...LEVELS[2]];

  } else {

    items = [...(LEVELS[levelVal] ?? [])];

  }

 

  const t = typeSel.value;

  if(t === "words") items = items.filter(x => x.kind === "word");

  if(t === "sentences") items = items.filter(x => x.kind === "sentence");

 

  // skapa "kort" som kan frågas i båda riktningar (eller vald riktning)

  const mode = modeSel.value;

  const cards = [];

 

  for(const it of items){

    if(mode === "fi2sv" || mode === "mix"){

      cards.push({ ...it, ask:"fi", answer:"sv" });

    }

    if(mode === "sv2fi" || mode === "mix"){

      cards.push({ ...it, ask:"sv", answer:"fi" });

    }

  }

 

  return shuffle(cards);

}

 

function showCard(){

  feedbackBox.style.display = "none";

  answerInput.value = "";

  answerInput.focus();

 

  current = deck[idx];

  if(!current){

    endQuiz();

    return;

  }

 

  currentDirection = (current.ask === "fi") ? "fi2sv" : "sv2fi";

 

  const prompt = current[current.ask];

  questionText.textContent = prompt;

 

  metaLine.innerHTML = "";

  const p1 = document.createElement("div");

  p1.className = "pill";

  p1.textContent = `Fråga ${idx+1} / ${deck.length}`;

  const p2 = document.createElement("div");

  p2.className = "pill";

  p2.textContent = current.kind === "word" ? "Ord" : "Mening";

  const p3 = document.createElement("div");

  p3.className = "pill";

  p3.textContent = currentDirection === "fi2sv" ? "FI → SV" : "SV → FI";

  const p4 = document.createElement("div");

  p4.className = "pill";

  p4.textContent = `Poäng: ${score}/${attempts}`;

  metaLine.append(p1,p2,p3,p4);

 

  document.getElementById("answerLabel").textContent =

    currentDirection === "fi2sv" ? "Skriv på svenska" : "Skriv på finska";

}

 

function grade(userInput){

  const correctRaw = current[current.answer];

  const accepted = splitAnswers(correctRaw);

 

  if(strictness === "strict"){

    for(const a of accepted){

      if(userInput.trim() === a) return { verdict:"ok", correct:a, note:"Rätt!" };

    }

    return { verdict:"bad", correct: accepted[0] ?? correctRaw, note:"Fel." };

  }

 

  // lenient

  for(const a of accepted){

    if(normalize(userInput) === normalize(a)) return { verdict:"ok", correct:a, note:"Rätt!" };

  }

  for(const a of accepted){

    if(isNear(userInput, a)) return { verdict:"near", correct:a, note:"Nära!" };

  }

  return { verdict:"bad", correct: accepted[0] ?? correctRaw, note:"Fel." };

}

 

function feedback(verdict, correct, user){

  const box = feedbackBox;

  box.style.display = "block";

  box.className = "fb " + (verdict === "ok" ? "ok" : verdict === "near" ? "near" : "bad");

 

  let msg = "";

  if(verdict === "ok") msg = "✅ Bra jobbat! Rätt svar.";

  if(verdict === "near") msg = "🟨 Nära! Kolla facit och skriv gärna en gång till i huvudet.";

  if(verdict === "bad") msg = "❌ Bra försök — här är rätt svar.";

 

  const extra = (currentDirection === "fi2sv")

    ? `Facit (sv): <b>${escapeHtml(correct)}</b>`

    : `Facit (fi): <b>${escapeHtml(correct)}</b>`;

 

  const tip = verdict !== "ok"

    ? `<div class="small muted" style="margin-top:8px">Tips: Fokusera på nyckelorden och ordföljd.</div>`

    : `<div class="small muted" style="margin-top:8px">Fortsätt så! 💪</div>`;

 

  box.innerHTML = `

    <div>${msg}</div>

    <div style="margin-top:8px">${extra}</div>

    <div class="small muted" style="margin-top:6px">Ditt svar: "${escapeHtml(user)}"</div>

    ${tip}

  `;

}

 

function escapeHtml(s){

  return (s ?? "").toString()

    .replaceAll("&","&amp;")

    .replaceAll("<","&lt;")

    .replaceAll(">","&gt;")

    .replaceAll('"',"&quot;");

}

 

function submit(){

  const user = answerInput.value;

  attempts++;

 

  const g = grade(user);

  if(g.verdict === "ok") score++;

  // "near" räknas inte som poäng, men du kan ändra om du vill

 

  feedback(g.verdict, g.correct, user);

 

  // gå vidare efter kort paus (så man hinner läsa). På mobil är detta skönt.

  setTimeout(() => {

    idx++;

    showCard();

  }, 1200);

}

 

function skip(){

  attempts++;

  feedback("bad", current[current.answer], "(hoppade över)");

  setTimeout(() => {

    idx++;

    showCard();

  }, 700);

}

 

function startQuiz(){

  strictness = strictSel.value;

  deck = buildDeck();

  idx = 0;

  score = 0;

  attempts = 0;

 

  if(deck.length === 0){

    alert("Inget innehåll valt. Lägg till data i LEVELS eller välj annan inställning.");

    return;

  }

 

  setupCard.style.display = "none";

  resultCard.style.display = "none";

  quizCard.style.display = "block";

 

  showCard();

}

 

function endQuiz(){

  quizCard.style.display = "none";

  resultCard.style.display = "block";

  const pct = attempts ? Math.round((score/attempts)*100) : 0;

  const msg =

    pct >= 90 ? "🔥 Super! Du är riktigt stark på det här."

    : pct >= 75 ? "💪 Bra nivå! Lite till så sitter det."

    : pct >= 50 ? "👍 Helt okej — fortsätt nöta så lossnar det."

    : "🌱 Bra att du tränar! Vi tar en runda till.";

 

  resultText.innerHTML = `

    <div>Poäng: <b>${score}</b> av <b>${attempts}</b> (${pct}%).</div>

    <div style="margin-top:8px">${msg}</div>

  `;

}

 

function resetAll(){

  idx = 0; score = 0; attempts = 0; deck = [];

  quizCard.style.display = "none";

  resultCard.style.display = "none";

  setupCard.style.display = "block";

}
